procesos 
  proceso contarPapeles(ES p:numero)
  comenzar 
    p:=0
    mientras(HayPapelEnLaEsquina)
      tomarPapel
      p:=p+1
    repetir p
      depositarPapel
  fin
  proceso contarFlores(ES f:numero)
  comenzar 
    f:=0
    mientras(HayFlorEnLaEsquina)
      tomarFlor
      f:=f+1
    repetir f 
      depositarFlor
  fin
  proceso cumpleCondicion(ES f:numero; ES p:numero; ES info:numero)
  comenzar 
    si ( p - f) = 1
      info:=info+1
    p:=0
    f:=0
  fin
  proceso izquierda
  comenzar
    repetir 3
      derecha
  fin
  proceso escalon
  variables
    f:numero
    p:numero
    info:numero
    pasos:numero
  comenzar
    info:=0
    pasos:=1
    repetir 4
      repetir pasos
        contarPapeles(p)
        contarFlores(f)
        mover
      derecha
      repetir pasos
        contarPapeles(p)
        contarFlores(f)
        mover
      izquierda
      cumpleCondicion(f,p,info)
      pasos:=pasos+1
    Informar(info)
  fin
