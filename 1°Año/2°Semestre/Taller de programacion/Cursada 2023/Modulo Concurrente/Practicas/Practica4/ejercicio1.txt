programa ejercicio1p4
procesos
  proceso juntarFlores
  comenzar
    mientras(HayFlorEnLaEsquina)    
      tomarFlor
  fin 
  proceso irAdepositar(E av:numero; E ca:numero; E f:numero)
  variables
    avOrigen:numero
    caOrigen:numero
  comenzar
    avOrigen:=PosAv
    caOrigen:=PosCa
    Pos(av,ca)
    repetir f
      depositarFlor
    Pos(avOrigen,caOrigen)
  fin
  proceso avanzar
  variables
    final:boolean
  comenzar
    final:=F
    mientras(HayFlorEnLaBolsa) & (final = F)
      si (PosCa <> 100) 
        depositarFlor
        mover
      sino
        final:=V
  fin
areas 
  ciudad:AreaC(1,1,100,100)
robots
  robot cliente
  variables
    ca:numero
    id:numero
    termine:boolean
    ok:boolean
    f:numero
    av:numero
  comenzar
    RecibirMensaje(id,jefe)
    termine := F
    mientras(termine = F)
      EnviarMensaje(id,jefe)
      Random(f,1,4)
      EnviarMensaje(f,jefe)
      av:=PosAv
      ca:=PosCa+1
      EnviarMensaje(av,jefe)
      EnviarMensaje(ca,jefe)
      RecibirMensaje(ok,jefe)
      Pos(av,ca)
      juntarFlores
      Pos(av,(ca - 1))
      avanzar
      si PosCa = 100
        termine := V
    EnviarMensaje(id,jefe)
    EnviarMensaje(0,jefe)
  fin
  robot fiscal
  variables
    quien:numero
    f:numero
    av:numero
    ca:numero
    termine:numero
  comenzar
    termine:=0
    EnviarMensaje(1,robot1)
    EnviarMensaje(2,robot2)
    EnviarMensaje(3,robot3)
    mientras (termine < 3)
      RecibirMensaje(quien,*)
      si (quien = 1)
        RecibirMensaje(f,robot1)
        si (f <> 0)
          RecibirMensaje(av,robot1)
          RecibirMensaje(ca,robot1)
          irAdepositar(av,ca,f)
          EnviarMensaje(V,robot1)
        sino 
          termine:=termine+1
      sino
        si (quien = 2)
          RecibirMensaje(f,robot2)
          si (f <> 0)
            RecibirMensaje(av,robot2)
            RecibirMensaje(ca,robot2)
            irAdepositar(av,ca,f)
            EnviarMensaje(V,robot2)
          sino
            termine:=termine+1
        sino
          RecibirMensaje(f,robot3)
          si (f <> 0)
            RecibirMensaje(av,robot3)
            RecibirMensaje(ca,robot3)
            irAdepositar(av,ca,f)
            EnviarMensaje(V,robot3)
          sino
            termine:=termine+1
  fin      
variables
  robot1:cliente
  robot2:cliente
  robot3:cliente
  jefe:fiscal
comenzar
  AsignarArea(robot1,ciudad)
  AsignarArea(robot2,ciudad)
  AsignarArea(robot3,ciudad)
  AsignarArea(jefe,ciudad)
  Iniciar(jefe,100,100)
  Iniciar(robot1,1,1)
  Iniciar(robot2,2,1)
  Iniciar(robot3,3,1)
fin
