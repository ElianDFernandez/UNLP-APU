programa ejercicio3p4
procesos
  proceso juntarPapeles
  comenzar
    mientras(HayPapelEnLaEsquina)
      tomarPapel
  fin
areas
  ciudad:AreaC(1,1,100,100)
robots
  robot limpiador
  variables
    id:numero
    aviso:numero
    termine:boolean
  comenzar
    RecibirMensaje(id,jefe)
    Informar(id)
    termine:=F
    mientras(termine = F)
      repetir 5
        juntarPapeles
        si (PosCa <> 100)
          mover
        sino
          termine:= V
      si id = 1
        EnviarMensaje(1,robot2)
        EnviarMensaje(1,robot3)
        RecibirMensaje(aviso,robot2)
        RecibirMensaje(aviso,robot3)
      sino 
        si id = 2
          EnviarMensaje(1,robot1)
          EnviarMensaje(1,robot3)
          RecibirMensaje(aviso,robot1)
          RecibirMensaje(aviso,robot3)
        sino
          EnviarMensaje(1,robot2)
          EnviarMensaje(1,robot1)
          RecibirMensaje(aviso,robot1)
          RecibirMensaje(aviso,robot2)
    juntarPapeles
  fin
  robot fiscal
  comenzar
    EnviarMensaje(1,robot1)
    EnviarMensaje(2,robot2)
    EnviarMensaje(3,robot3)
  fin
variables
  robot1:limpiador
  robot2:limpiador
  robot3:limpiador
  jefe:fiscal
comenzar
  AsignarArea(robot1,ciudad)
  AsignarArea(robot2,ciudad)
  AsignarArea(robot3,ciudad)
  AsignarArea(jefe,ciudad)
  Iniciar(robot1,1,1)
  Iniciar(robot2,2,1)
  Iniciar(robot3,3,1)
  Iniciar(jefe,4,1)
fin
