programa parcial1
procesos 
  proceso juntarPapeles(ES papeles:numero)
  comenzar 
    mientras(HayPapelEnLaEsquina) 
      tomarPapel
      papeles:=papeles+1
  fin
  proceso depositarPapeles(ES papeles:numero)
  comenzar 
    repetir papeles
      depositarPapel
  fin
areas 
  comienzo1parte1:AreaP(1,1,7,29)
  comienzo1parte2:AreaP(2,30,7,30)
  comienzo1parte3:AreaP(1,31,7,100)
  comienzo2:AreaP(8,1,14,100)
  comienzo3:AreaP(15,1,21,100)
  areaFiscal:AreaP(25,1,25,1)
  deposito:AreaPC(1,30,1,30)
robots 
  robot corredor 
  variables 
    id,pasos,papeles,avOrigen,caOrigen,aux:numero
    avInicio:numero
    caInicio:numero
    pasosTotal:numero
  comenzar 
    RecibirMensaje(id,fiscalizador)
    RecibirMensaje(avInicio,fiscalizador)
    RecibirMensaje(caInicio,fiscalizador)
    Pos(avInicio,caInicio)
    pasosTotal:=0
    repetir 5
      Random(pasos,1,10)
      repetir pasos 
        juntarPapeles(papeles)
        si (PosAv = 1)
          si (PosCa = 29)
            BloquearEsquina(1,30)
            mover
          sino
            si (PosCa = 30)
              mover
              LiberarEsquina(1,30)
        sino      
          mover
        pasosTotal:=pasosTotal+pasos
    si (PosAv = 1) && (PosCa = 30)
      mover
      LiberarEsquina(1,30)
    avOrigen:=PosAv
    caOrigen:=PosCa
    BloquearEsquina(1,30)
    Pos(1,30)
    depositarPapeles(papeles)
    Pos(avOrigen,caOrigen)
    LiberarEsquina(1,30)
    EnviarMensaje(id,fiscalizador)
    EnviarMensaje(pasosTotal,fiscalizador)
  fin
  robot fiscal
  variables
    avInicio:numero
    caInicio:numero
    quien:numero
    pasos:numero
    pasosGanador:numero
    robotGanador:numero
  comenzar
    EnviarMensaje(1,robot1)
    EnviarMensaje(2,robot2)
    EnviarMensaje(3,robot3)
    Random(avInicio,1,7)
    Random(caInicio,1,40)
    EnviarMensaje(avInicio,robot1)
    EnviarMensaje(caInicio,robot1)
    Random(avInicio,8,14)
    Random(caInicio,1,40)
    EnviarMensaje(avInicio,robot2)
    EnviarMensaje(caInicio,robot2)
    Random(avInicio,15,21)
    Random(caInicio,1,40)
    EnviarMensaje(avInicio,robot3)
    EnviarMensaje(caInicio,robot3)
    pasosGanador:=0
    repetir 3
      RecibirMensaje(quien,*)
      si quien = 1 
        RecibirMensaje(pasos,robot1)
        si pasos > pasosGanador
          pasosGanador := pasos
          robotGanador := 1
      sino
        si quien = 2
          RecibirMensaje(pasos,robot2)
          si pasos> pasosGanador
            pasosGanador := pasos
            robotGanador := 2
        sino
          RecibirMensaje(pasos,robot3)
          si pasos > pasosGanador
            pasosGanador := pasos
            robotGanador := 3
    Informar(robotGanador)
  fin
variables
  robot1:corredor
  robot2:corredor
  robot3:corredor
  fiscalizador:fiscal
comenzar
  AsignarArea(robot1,comienzo1parte1)

  AsignarArea(robot2,comienzo2)
  AsignarArea(robot3,comienzo3)
  AsignarArea(fiscalizador,areaFiscal)
  AsignarArea(robot1,deposito)
  AsignarArea(robot2,deposito)
  AsignarArea(robot3,deposito)
  Iniciar(robot1,1,1)
  Iniciar(robot2,8,1)
  Iniciar(robot3,15,1)
  Iniciar(fiscalizador,25,1)
fin
