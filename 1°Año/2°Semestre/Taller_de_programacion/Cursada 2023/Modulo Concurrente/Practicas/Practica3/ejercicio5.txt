  proceso avanzar(ES p:numero; ES termine:boolean)
  variables
    avOrigen:numero
    caOrigen:numero
  comenzar
    avOrigen:=PosAv
    caOrigen:=PosCa
    BloquearEsquina(11,11)
    Pos(11,11)
    si (HayPapelEnLaEsquina)
      tomarPapel
      Pos(avOrigen,caOrigen)
      LiberarEsquina(11,11)
      depositarPapel
      mover
      p:=p+1
    sino
      termine:=V
      Pos(avOrigen,caOrigen)
      LiberarEsquina(11,11)
  fin

  robot fiscal 
  variables 
    pos1:numero
    pos2:numero
    pos3:numero
    pos4:numero
    quien:numero
  comenzar
    EnviarMensaje(1,robot1)
    EnviarMensaje(2,robot2)
    EnviarMensaje(3,robot3)
    EnviarMensaje(4,robot4)
    repetir 4 
      RecibirMensaje(quien,*)
      si quien = 1
        RecibirMensaje(pos1,robot1)
      sino
        si quien = 2
          RecibirMensaje(pos2,robot2)
        sino
          si quien = 3
            RecibirMensaje(pos3,robot3)
          sino
            RecibirMensaje(pos4,robot4)
    si (pos1 > pos2) & (pos1 > pos3) & (pos1 > pos4)
      Informar(1)
    sino
      si (pos2 > pos1) & (pos2 > pos3) & (pos2 > pos4)
        Informar(2)
      sino 
        si (pos3 > pos1) & (pos3 > pos2) & (pos3 > pos4)
          Informar(3)
        sino 
          Informar(4)
  fin
