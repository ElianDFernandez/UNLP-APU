procesos 
  proceso moverFlores
  variables
    avOrigen:numero
    caOrigen:numero
  comenzar
    avOrigen:=PosAv
    caOrigen:=PosCa
    BloquearEsquina(10,10)
    Pos(10,10)
    mientras(HayFlorEnLaEsquina)
      tomarFlor
      Pos(avOrigen,caOrigen)
      LiberarEsquina(10,10)
      BloquearEsquina(11,11)
      Pos(11,11)
      depositarFlor
      Pos(avOrigen,caOrigen)
      LiberarEsquina(11,11)
      BloquearEsquina(10,10)
      Pos(10,10)
    Pos(avOrigen,caOrigen)
    LiberarEsquina(10,10)
  fin
  proceso recolectarFlores
  comenzar
    Pos(10,10)
    mientras(HayFlorEnLaEsquina)
      tomarFlor
    Pos(1,1)
  fin
