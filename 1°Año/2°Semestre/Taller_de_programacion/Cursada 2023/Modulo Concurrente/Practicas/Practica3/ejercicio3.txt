  proceso limpiarEsquina(E av:numero; E ca:numero; ES f:numero)
  variables
    avOrigen:numero
    caOrigen:numero
  comenzar 
    avOrigen:=PosAv
    caOrigen:=PosCa
    f:=0
    BloquearEsquina(av,ca)
    Pos(av,ca)
    mientras (HayFlorEnLaEsquina)
      tomarFlor
      f:=f+1
      Pos(avOrigen,caOrigen)
      LiberarEsquina(av,ca)
      BloquearEsquina(av,ca)
      Pos(av,ca)
    Pos(avOrigen,caOrigen)
    LiberarEsquina(av,ca)
  fin

    EnviarMensaje(1,robot1)

    EnviarMensaje(4,robot4)
    Random(av,2,10)

    EnviarMensaje(av,robot1)
    EnviarMensaje(ca,robot1)
    EnviarMensaje(av,robot2)
    EnviarMensaje(ca,robot2)

    EnviarMensaje(ca,robot3)    
    EnviarMensaje(av,robot4)
    EnviarMensaje(ca,robot4)
    repetir 4
      RecibirMensaje(quien,*)
      si quien = 1 
        RecibirMensaje(tot1,robot1)
      sino 
        si quien = 2 
          RecibirMensaje(tot2,robot2)
        sino 
          si quien = 3 
            RecibirMensaje(tot3,robot3)
          sino
            RecibirMensaje(tot4,robot4)
    si (tot1 > tot2) & (tot1 > tot3) & (tot1 > tot4)
      Informar(1)
    sino  
      si (tot2 > tot1) & (tot2 > tot3) & (tot2 > tot4) 
        Informar(2)
      sino
        si (tot3 > tot1) & (tot3 > tot2) & (tot3 > tot4)
          Informar(3)
        sino
          Informar(4)
  fin
