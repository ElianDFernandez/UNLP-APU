procesos 
  proceso avanzarJuntando(ES p:numero; ES termine:boolean)
  comenzar
    mientras(p > 0) & (termine = F)
      mientras(HayPapelEnLaEsquina)
        p:= p - 1
        tomarPapel
      mover
      si (PosCa = 100)
        termine := V
  fin
  proceso avisar(E id:numero; ES terminaron:numero; E termine:boolean; E avisos:boolean)
  variables
    reps:numero
  comenzar
    si id = 1 
      EnviarMensaje(termine,robot2)
      EnviarMensaje(termine,robot3)
    sino
      si id = 2
        EnviarMensaje(termine,robot1)
        EnviarMensaje(termine,robot3)
      sino
        si id = 3
          EnviarMensaje(termine,robot1)
          EnviarMensaje(termine,robot2)
    repetir ( 2 - terminaron)
      RecibirMensaje(avisos,*)
      si (avisos = V)
        terminaron := terminaron + 1 
  fin
