programa eje3
procesos 
  proceso avanzarJuntando(ES p:numero)
  variables
    juntados:numero
  comenzar
    juntados:=0
    mientras (juntados < p) & (PosCa < 100)
      mientras (HayPapelEnLaEsquina) & (juntados < p)
        tomarPapel
        juntados:=juntados+1
      mover
    si (PosCa = 100) & (juntados < p)
      mientras(HayPapelEnLaEsquina) & (juntados < p)
        tomarPapel
        juntados:=juntados + 1
  fin
areas
  ave1:AreaP(1,1,1,100)
  ave2:AreaP(2,1,2,100)
  ave3:AreaP(3,1,3,100)
  aF:AreaP(100,100,100,100)  
robots
  robot fiscalizador
  comenzar
    EnviarMensaje(1,r1)
    EnviarMensaje(2,r2)
    EnviarMensaje(3,r3)
  fin
  robot papelero
  variables
    id:numero
    p:numero
    termine:boolean
    terminaron:numero
    termino:boolean
  comenzar
    RecibirMensaje(id,fiscalizador)
    terminaron:=0
    mientras PosCa<100
      termine:=F
      Random(p,1,5)
      avanzarJuntando(p)
      si (id=1)
        EnviarMensaje(termine,r2)
        EnviarMensaje(termine,r3)
        p:=0
      sino
        si (id=2)
          EnviarMensaje(termine,r1)
          EnviarMensaje(termine,r3)
          p:=0
        sino
          EnviarMensaje(termine,r1)
          EnviarMensaje(termine,r2)
          p:=0
      repetir (2 - terminaron)
        RecibirMensaje(termino,*)
        si (termino = V)
          terminaron:=terminaron+1
    termine:=V
    si (PosCa=100)
      si (id=1)
        EnviarMensaje(termine,r2)
        EnviarMensaje(termine,r3)
      sino
        si (id=2)
          EnviarMensaje(termine,r1)
          EnviarMensaje(termine,r3)
        sino
          EnviarMensaje(termine,r1)
          EnviarMensaje(termine,r2)        
  fin
variables
  r1:papelero
  r2:papelero
  r3:papelero
  fiscalizador:fiscalizador
comenzar
  AsignarArea(r1,ave1)
  AsignarArea(r2,ave2)
  AsignarArea(r3,ave3)
  AsignarArea(fiscalizador,aF) 
  Iniciar(r1,1,1)
  Iniciar(r2,2,1) 
  Iniciar(r3,3,1) 
  Iniciar(fiscalizador,100,100)  
fin
