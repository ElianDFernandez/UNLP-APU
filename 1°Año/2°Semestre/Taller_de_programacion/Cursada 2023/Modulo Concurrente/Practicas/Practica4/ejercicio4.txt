programa ejercicio4p4
procesos 
  proceso tarea(E tipo:numero; E av:numero; E ca:numero)
  variables
    avOrigen:numero
    caOrigen:numero
  comenzar
    avOrigen:=PosAv
    caOrigen:=PosCa
    si (tipo = 1)
      BloquearEsquina(av,ca)
      Pos(av,ca)
      mientras(HayFlorEnLaEsquina)
        tomarFlor
      Pos(avOrigen,caOrigen)
      LiberarEsquina(av,ca)
    sino
      si (tipo = 2)
        BloquearEsquina(av,ca)
        Pos(av,ca)
        mientras(HayPapelEnLaEsquina)
          tomarPapel
        Pos(avOrigen,caOrigen)
        LiberarEsquina(av,ca)
      sino
        si (tipo = 3)
          BloquearEsquina(av,ca)
          Pos(av,ca)
          mientras(HayFlorEnLaBolsa)
            depositarFlor
          mientras(HayPapelEnLaBolsa)
            depositarPapel
          Pos(avOrigen,caOrigen)
          LiberarEsquina(av,ca)
  fin
areas
  areaJ: AreaP(1,1,1,1)
  area1: AreaP(2,1,2,1)
  area2: AreaP(3,1,3,1)
  cuadrante: AreaP(2,2,100,100)
robots
  robot trabajador
  variables
    tareaTipo:numero
    av:numero
    ca:numero
  comenzar
    RecibirMensaje(tareaTipo,jefe)
    mientras(tareaTipo <> 4)
      RecibirMensaje(av,jefe)
      RecibirMensaje(ca,jefe)
      tarea(tareaTipo,av,ca)
      RecibirMensaje(tareaTipo,jefe)
  fin
  robot fiscal
  variables
    trabajador:numero
    tarea:numero
    t1:boolean
    t2:boolean
    av:numero
    ca:numero
  comenzar
    t1:=V
    t2:=V
    repetir 10 
      si (t1) & (t2)
        Random(trabajador,1,2)
      sino
        si (t1 = F) & (t2 = V)
          trabajador := 2
        sino
          trabajador := 1 
      Random(tarea,1,4)
      Random(av,2,100)
      Random(ca,2,100)
      si(trabajador = 1) 
        si(tarea = 4) 
          t1:=F
          EnviarMensaje(4,robot1)
        sino
          EnviarMensaje(tarea,robot1)
          EnviarMensaje(av,robot1)
          EnviarMensaje(ca,robot1)
      sino
        si (tarea = 4)
          t2:=F
          EnviarMensaje(4,robot2)
        sino
          EnviarMensaje(tarea,robot2)
          EnviarMensaje(av,robot2)
          EnviarMensaje(ca,robot2)
    si(t1)
      EnviarMensaje(4,robot1)
    si (t2)
      EnviarMensaje(4,robot2)
  fin
variables
  robot1:trabajador
  robot2:trabajador
  jefe:fiscal
comenzar
  AsignarArea(robot1,area1)

  AsignarArea(robot2,cuadrante)
  Iniciar(jefe,1,1)
  Iniciar(robot1,2,1)
  Iniciar(robot2,3,1)
fin
